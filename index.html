const MIN_LIQUIDITY = 0; // <<-- temporarily 0 for launch / debug

function debugLog(...args){
  try{
    const s = args.map(a => (typeof a === 'string' ? a : JSON.stringify(a, null, 2))).join(' ');
    const now = new Date().toLocaleTimeString();
    debugPanel.textContent = `[${now}] ` + s + "\n" + debugPanel.textContent;
  }catch(e){}
}

// safe number parser (removes commas/other thousands separators)
function safeNum(v){
  if (v === null || v === undefined) return 0;
  const s = String(v).replace(/[^\d.\-eE+]/g,'');
  const n = Number(s);
  return isFinite(n) ? n : 0;
}

async function fetchTokens(){
  apiStatusEl.textContent = 'Fetching...';
  debugLog("Starting fetch to " + ENDPOINT);
  try{
    const res = await fetch(ENDPOINT, { headers: { "accept": "application/json" }});
    debugLog("Response status:", { ok: res.ok, status: res.status, headers: (()=>{ const o={}; for(const k of res.headers.keys()) o[k]=res.headers.get(k); return o; })() });
    if (!res.ok) {
      apiStatusEl.textContent = `Proxy: error ${res.status}`;
      const text = await res.text().catch(()=>'[no body]');
      tokenContainer.innerHTML = `
Proxy returned ${res.status} ‚Äî cek log
`;
      debugLog("Non-ok body:", text);
      lastUpdateEl.textContent = new Date().toLocaleTimeString();
      return;
    }

    const data = await res.json().catch(()=>null);
    debugLog("Parsed JSON:", data && (data.result ? 'result array length=' + data.result.length : Object.keys(data)));
    const items = Array.isArray(data && data.result) ? data.result : [];
    window.tokenMap = {};
    tokenContainer.innerHTML = '';

    // apply sanitization & filter
    const filtered = items.filter(it => {
      const liq = safeNum(it.liquidity || it.liquid || it.liquidity_usd || 0);
      return liq >= MIN_LIQUIDITY;
    });

    if (!filtered.length) {
      tokenContainer.innerHTML = `
No tokens passed liquidity filter (>= ${MIN_LIQUIDITY}).
`;
    } else {
      filtered.forEach(it => {
        // normalize address field
        const addr = it.tokenAddress || it.address || ('addr_' + Math.random().toString(36).slice(2,9));
        window.tokenMap[addr] = it;

        const liquidity = safeNum(it.liquidity || it.liquid || 0);
        const fdv = safeNum(it.fullyDilutedValuation || it.fdv || 0);
        const createdAt = it.createdAt ? new Date(it.createdAt).toLocaleTimeString() : '‚Äî';

        const div = document.createElement('div');
        div.className = 'token';
        div.dataset.addr = addr;
        div.innerHTML = `
          

            
${(it.symbol || '?')} ‚Äî ${it.name || 'Unnamed'}

            
üíß $${isFinite(liquidity) ? liquidity.toLocaleString() : '‚Äî'} ‚Ä¢ üßÆ FDV: $${isFinite(fdv) ? fdv.toLocaleString() : '‚Äî'}

          

          
${createdAt}

        `;
        tokenContainer.appendChild(div);
      });
    }

    apiStatusEl.textContent = 'Proxy: OK';
    lastUpdateEl.textContent = new Date().toLocaleTimeString();
  } catch (err){
    console.error(err);
    debugLog("Fetch error:", (err && (err.stack||err.message||err)));
    apiStatusEl.textContent = 'Proxy: error';
    tokenContainer.innerHTML = `
Network error ‚Äî check proxy & console.
`;
    lastUpdateEl.textContent = new Date().toLocaleTimeString();
  }
}

async function loadFeaturedToken(){
  const addr = FEATURED_TOKEN_ADDRESS || null;
  const titleEl = document.getElementById('featuredTitle');
  const wolfData = document.getElementById('wolfData');
  const logoEl = document.getElementById('featuredLogo');
  const buyBtn = document.getElementById('btnBuy');
  const solBtn = document.getElementById('btnSolscan');

  if (!addr || addr === "YOUR_TOKEN_ADDRESS") {
    wolfData.textContent = 'Featured token not set. Set FEATURED_TOKEN_ADDRESS in index.html after launch.';
    buyBtn.href = '#'; solBtn.href = '#';
    return;
  }

  buyBtn.href = `https://pump.fun/coin/${addr}`;
  solBtn.href = `https://solscan.io/token/${encodeURIComponent(addr)}?cluster=mainnet`;

  try{
    const res = await fetch(FEATURE_PRICE_URL(addr), { headers: { "accept":"application/json" }});
    if (!res.ok) { wolfData.textContent = `No live price (status ${res.status})`; return; }
    const data = await res.json();
    if (data && data.usdPrice) {
      wolfData.innerHTML = `üí∞ Price: $${Number(data.usdPrice).toFixed(6)} ‚Ä¢ ‚è± ${new Date().toLocaleTimeString()}`;
      const liquidity = safeNum(data.liquidity || 0);
      const fdv = safeNum(data.fullyDilutedValuation || 0);
      const strength = Math.min(95, Math.round(Math.min(100, (liquidity / Math.max(1, fdv)) * 100 * 2)));
      document.getElementById('howlFill').style.width = Math.max(6, strength) + '%';
      logoEl.src = data.logo || logoEl.src;
    } else { wolfData.textContent = 'No price data'; }
  }catch(e){ console.error(e); wolfData.textContent = 'Error loading featured data'; }
}

let paused = false;
let pollHandle = null;
function startPolling(){ stopPolling(); pollHandle = setInterval(fetchTokens, POLL_INTERVAL); }
function stopPolling(){ if (pollHandle) clearInterval(pollHandle); pollHandle = null; }

refreshBtn.addEventListener('click', fetchTokens);
pauseBtn.addEventListener('click', () => {
  paused = !paused;
  pauseBtn.textContent = paused ? 'Resume' : 'Pause';
  if (!paused) startPolling(); else stopPolling();
});
